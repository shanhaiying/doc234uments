\qu (15分) 对给定线性方程组 $\left\{ \begin{array}{ll} x_1+x_2+a\ x_3 &=4 \\
x_1-x_2+2x_3 &=-4\\-x_1+a\ x_2+x_3 &=a^2 \end{array} \right.$
\begin{parts}
\part 当$a$取何值时，方程组有唯一解；
\part 当$a$取何值时，方程组无解；
\part 当$a$取何值时，方程组有无穷多解，并求其通解。
\end{parts}
\begin{sol}
$\begin{pmatrix}1 & 1 & a & | & 4 \\ 1 & -1 & 2 & | & 4\\
-1 & 1 & 1 & | & a^2\end{pmatrix} \Rightarrow \begin{pmatrix}
1 & 1 & a & | & 4 \\ 0 & -2 & 2-a & | & -8 \\
0 & 0 & \dfrac{(a+1)(4-a)}{2} & | & a^2-4a \end{pmatrix}$
\begin{parts}
\part $a\neq-1$且$a\neq4$
\part $a=-1$
\part $a=4$，$\begin{pmatrix}1 & 1 & 4 &|& 4\\
0 & -2 & -2 &|& -8 \\ 0 & 0 & 0 &|& 0\end{pmatrix}$，特解为
$\begin{pmatrix}-12\\0\\4\end{pmatrix}$，齐次方程解为
$\begin{pmatrix}-3\\-1\\1\end{pmatrix}\lambda$
\end{parts}
\end{sol}

\part 求直线$L$与平面$\pi$的夹角。

\begin{parts}

\qu (8分) 设复方阵$A$满足$A'A=I$，$\lambda_0$是$A$的一个特征值，\\
证明：$\lambda_0\neq0$，且$\lambda_0^{-1}$也是$A$的一个特征值。
\begin{sol}
$A'A=I\Rightarrow A^{-1}=A'$，$\lambda_0$为$A$的特征值，
所以有$Ax_0=\lambda_0x_0$，即$A^{-1}x_0=\lambda_0^{-1}x_0$，
即$\lambda_0^{-1}$是$A^{-1}$也就是$A'$的特征值，因此是$A$的特征值。
\end{sol}

\qu (15分) 用正交变换$X=PY$将二次型$Q(x)=x_1^2+3x_2^2+x_3^2+4x_1x_3$化为标准形
\begin{sol}
$A=\begin{pmatrix}1 & 0 & 2\\0 & 3 & 0\\2 & 0 & 1\end{pmatrix}$，可得
$|\lambda I-A|=(\lambda-3)^2(\lambda+1)$，有$\lambda_0=3$，$\lambda_1=-1$。
对$\lambda_0=3$，有$x_1=\begin{pmatrix}0\\1\\0\end{pmatrix}$和
$x_2=\begin{pmatrix}1\\0\\1\end{pmatrix}$，对$\lambda_1=-1$，有
$x_3=\begin{pmatrix}-1\\0\\1\end{pmatrix}$，所以有
$P=\begin{pmatrix}0 & \sqrt2/2 & -\sqrt2/2\\1 & 0 & 0\\
0 & \sqrt2/2 & \sqrt2/2\end{pmatrix}$
\end{sol}

\qu (10分) 已知$3$阶方阵$A$与$3$维列向量$x$使得向量组$x,Ax,A^2x$线性无关，\\
且满足$A^3x=3x-Ax+2A^2x$，记$P=(x,Ax,A^2x)$，
\begin{parts}
\part 求$3$阶方阵$B$使$A=PBP^{-1}$；
\part 计算$det(A+I)$。
\end{parts}
\begin{sol}
\begin{parts}
\part 由$AP=PB$，即$A(x,Ax,A^2x)=(x,Ax,A^2x)B$，又\\
$Ax=(x,Ax,A^2x)\begin{pmatrix}0\\1\\0\end{pmatrix}$，
$AAx=(x,Ax,A^2x)\begin{pmatrix}0\\0\\1\end{pmatrix}$，
$AA^2x=(x,Ax,A^2x)\begin{pmatrix}3\\-1\\2\end{pmatrix}$，
所以$B=\begin{pmatrix}0 & 0 & 3\\1 & 0 & -1\\0 & 1 & 2\end{pmatrix}$
\part $det(A+I)=det(P(A+I)P^{-1})=det(B+I)=det\begin{pmatrix}
1 & 0 & 3\\1 & 1 & -1\\0 & 1 & 3\end{pmatrix}$
\end{parts}
\end{sol}
