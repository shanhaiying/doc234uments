$\left\{\begin{array}{l} \dfrac{dy}{dx}=f(x,y) \\ y(a)=y_0\end{array}\right.

$$y(x_{n+1})=y(x_{n-p})+\int_{x_{n-p}}^{x_{n+1}}y'(x)dx$$

\end{enumerate}
