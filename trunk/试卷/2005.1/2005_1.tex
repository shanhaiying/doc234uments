\qu  (12分) 构造线性多步法$p=3,q=2$的隐式差分格式。

$\{x_{n+1},x_n,x_{-1}\}$ \\

 \dfrac{(x-x_n)(x-x_{n-1})}{(x_{n+1}-x_n)(x_{n+1}-x_{n-1})}dx
 =\dfrac{8}{3}h$$
(3分)$$\int_{x_{n-3}}^{x_{n+1}}
 \dfrac{(x-x_{n+1})(x-x_{n-1})}{(x_{n}-x_{n+1})(x_{n}-x_{n-1})}dx
 =\dfrac{-16}{3}h$$

$$y_{n+1}=y_{n-3}+\dfrac{8h}{3}f(x_{n+1},y_{n+1})

$$
\end{sol}
