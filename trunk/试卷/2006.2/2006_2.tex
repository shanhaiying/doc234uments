\begin{parts}
\part 分析该方程在区间$[1,2]$上的根的情况；
\part 用Newton迭代法，取初值$x_0=1.5$，求出该方程的根。(最多迭代3步)
\end{parts}
\begin{sol}
$f(x)=x+ln(x)-2$，则$f(1)=-1<0,f(2)=ln(2)>0$ \rfen{2} ，
且$g'(x)=1+1/x>0,x\in[1,2]$ \rfen{2} ，
所以有1个根 \rfen{1} \\
$x_{k+1}=x_k-\dfrac{x_k+ln(x_k)-2}{1+1/x_k}$
\begin{tabular}{|c|l|}
\hline
i & $x_i$  \\
\hline
0 & 1.5 \\
1 & 1.5567 \\
2 & 1.5572 \\
3 & 1.5572 \\
\hline
\end{tabular} \\
根为$1.5572$ \rfen{5}
\end{sol}

\qu (15分)对矩阵
$$\begin{pmatrix}3 & 1 & 0 \\ 2 & 4 & t \\ 0 & t & 3 \end{pmatrix}$$
\begin{parts}
\part 求出$t$的范围，使其Jacobi迭代格式收敛
\part 求出$t$的范围，使其Gauss-Siedel迭代格式收敛
\end{parts}
\begin{sol}
Jacobi 迭代的迭代矩阵为
$\begin{pmatrix}0 & -1/3 & 0 \\ -1/2 & 0 & -t/4 \\ 0 & -t/3 & 0
\end{pmatrix}$ \rfen{3} \\
特征多项式为$\lambda(\lambda^2-t^2/12-1/6)$，特征值为$0,\pm\sqrt{t^2/12+1/6}$
\rfen{2} \\
$-\sqrt{10}<t<\sqrt{10}$ \rfen{2}\\
Gauss-Seidel 迭代的迭代矩阵为
$\begin{pmatrix}0 & 1/3 & 0 \\ 0 & -1/6 & t/4 \\ 0 & t/18 & -t^2/12
\end{pmatrix}$ \rfen{4} \\
特征多项式为$\lambda^2(\lambda^2+t^2/12+1/6)$，特征值为$0,-t^2/12-1/6$
\rfen{2} \\
$-\sqrt{10}<t<\sqrt{10}$ \rfen{2}
\end{sol}
