\pingfen{二、解答题} \qu (10分) 给定方程 $x+ln(x)=2$
\begin{parts}
\part 分析该方程在区间$[1,2]$上的根的情况；
\part 用Newton迭代法，取初值$x_0=1.5$，求出该方程的根。(精确到小数点后4位)
\end{parts}
\begin{sol}
$f(x)=x+ln(x)-2$，则$f(1)=-1<0,f(2)=ln(2)>0$ \rfen{2} ，
且$g'(x)=1+1/x>0,x\in[1,2]$ \rfen{2} ，
所以有1个根 \rfen{1} \\
$x_{k+1}=x_k-\dfrac{x_k+ln(x_k)-2}{1+1/x_k}$
\begin{tabular}{|c|l|}
\hline
i & $x_i$  \\
\hline
0 & 1.5 \\
1 & 1.5567 \\
2 & 1.5572 \\
3 & 1.5572 \\
\hline
\end{tabular} \\
根为$1.5572$ \rfen{5}
\end{sol}
